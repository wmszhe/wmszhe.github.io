fun binarySearch(aar: IntArray, key: Int): Int {
    var low: Int = 0
    var high: Int = aar.size - 1
    if (aar[low] > key || aar[high] < key || low > high) {
        return -1
    }
    var mid: Int = 0
    while (low <= high) {
        mid = (low + high) / 2
        when {
            aar[mid] < key -> low = mid + 1
            aar[mid] > key -> high = mid - 1
            else -> return mid
        }
    }
    return -1
}
