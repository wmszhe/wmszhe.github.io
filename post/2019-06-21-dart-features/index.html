String say(String from, String msg, [String device, String time]) {

      result = "$result with a $device";

    if (time != null) {
      result = "$result at $time";
    }
 
    return result;
}

    // david says hello with a 2019.2.29.20:08:08 at mobile
    print(say("david", "hello"));
    // david says hello
}
